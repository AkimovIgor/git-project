// Инициализация проекта (директории)
git init

// Конфигурация и установка имени пользователя
git config --global user.name "my_name"

// Конфигурация и установка email пользователя
git config --global user.email "my-email@gmail.com"

// Получить список конфигурации
git config --list

// Для добавления каталогов и файлов в список игнорируемых нужно создать в корне файл .gitignore.
// В этом файле нужно указать, какие каталоги или файлы будут проигнорированы, например:
	
logs/
docs/*.txt


// Показывает добавленные (измененные) файлы и папки
git status

// Более подробно показывает добавленные/измененные файлы и папки
git status --untracked-files=all

// Инициализирует и подготавливает добавленные/измененные все файлы перед коммитом
git add .

// Инициализирует и подготавливает добавленный/измененный файл перед коммитом
git add index.php

// Инициализирует и подготавливает добавленные/измененные все файлы с расширением .php перед коммитом
git add "*.php"

// Удаляет файл из списка подготовленных для коммита
git rm --cached index.php

// Первоначальный коммит файлов
git commit -a -m"init"

// Коммит файлов
git commit -m"comment"

// Откат закомиченных изменений
git checkout index.php

// Отображение списка всех коммитов
git log

// Отображение списка всех коммитов за последние 2 недели
git log --since=2.weeks

// Красивое отображение списка всех коммитов
git log --graph --all --decorate --oneline -10 --pretty=format:"%h - %an, %ar : %s"

// Показывает изменения в последних 2-х коммитах
git log -p -2

// Создает новую ветку и переходит в неё
git checkout -b new_f

// Отображение всех веток
git branch

// Отображение всех веток с последними коммитами
git branch -v

// Создание ветки (без перехода к ней)
git branch new_f2

// Переход в указанную ветку
git checkout new_f2

// Запуск утилиты для мерджа(слияния/соединения) веток
git mergetool

/******************** РАБОТА С GITHUB ************************/

// Подключаем репозиторий на GitHub к проекту
git remote add origin https://github.com/AkimovIgor/git-project.git

// Добавлление закомиченных данных в репозиторий на GitHub
git push -u origin master
