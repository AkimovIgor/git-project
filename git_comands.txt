git status                                                                              // Показывает добавленные (измененные) файлы и папки
git status --untracked-files=all                                              // Более подробно показывает добавленные (измененные) файлы и папки
git add .                                                                                // Инициализирует и подготавливает добавленные/измененные все файлы перед коммитом
git add index.php                                                                  // Инициализирует и подготавливает добавленный/измененный файл перед коммитом
git add "*.php"                                                                      // Инициализирует и подготавливает добавленные/измененные все файлы с расширением .php перед коммитом
git rm --cached index.php                                                      // Удаляет файл из списка подготовленных для коммита
git commit -a -m"init"                                                             // Коммит файлов
git checkout index.php                                                           // Откат закомиченных изменений
git log                                                                                   // Отображение списка всех коммитов
git log --since=2.weeks                                                          // Отображение списка всех коммитов за последние 2 недели
git log --graph --all --decorate --oneline -10 --pretty=format:"%h - %an, %ar : %s"                                // Красивое отображение списка всех коммитов
git log -p -2                                                                           // Показывает изменения в последних 2-х коммитах
git checkout -b new_f                                                           // Создает новую ветку и переходит в неё
git branch                                                                             // Отображение всех веток
git branch -v                                                                         // Отображение всех веток с последними коммитами
